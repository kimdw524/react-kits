
<Typography>
  When using `Next.js`, use `next-themes` with the class attribute.
</Typography>

```tsx

export const Providers = ({ children }: { children: React.ReactNode }) => (
  <ThemeProvider attribute="class">{children}</ThemeProvider>
);
```

      <Typography as="h3">UIProvider</Typography>
      <Typography>
        Wrap your application with `UIProvider` to support overlays.
      </Typography>

```tsx
import { StrictMode } from 'react';
import { createRoot } from 'react-dom/client';

import { UIProvider } from '@kimdw-rtk/ui';
import '@kimdw-rtk/ui/css/default-style';
import '@kimdw-rtk/ui/css/reset';

createRoot(document.getElementById('root')!).render(
  <StrictMode>
    <UIProvider container={document.body}>
      <App />
    </UIProvider>
  </StrictMode>,
);
```

      <Typography as="h3">Utility classes</Typography>
      <Typography>
        Use the `sx` function to generate utility classes. This allows you to
        apply styles based on design tokens as well as responsive values.
      </Typography>

```tsx
import { Box, sx } from '@kimdw-rtk/ui';

// Use sx directly in className for HTML elements.
<div
  className={sx({
    display: 'flex',
    paddingX: { mobile: 'md', desktop: 'xl' },
  })}
>
  Div
</div>;

// Use the sx prop with UI components.
<Box
  sx={{
    display: 'flex',
    paddingX: { mobile: 'md', desktop: 'xl' },
  }}
>
  Box
</Box>;
```
